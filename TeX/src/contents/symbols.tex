% \nomenclature{}{}

\nomenclature{$M$}{Matrix}
\nomenclature{$\mathbf{m}$}{Vector}
\nomenclature{$m$}{scalar}

\nomenclature{$f(x;\; c)$}{Function $f$, which depends on variable $x$ and constant $c$}
\nomenclature{$\mathbb{R}^3$}{Three-dimensional space}
\nomenclature{$\Omega$}{A volume in space}
\nomenclature{$\Gamma$}{A surface in space}
\nomenclature{$\Gamma=\partial \Omega$}{A boundary (surface) of a volume}
\nomenclature{$\gamma$}{A curve in space}
\nomenclature{${\bf x}$}{A point in space}
\nomenclature{${\bf r}$}{An observation point in space (a.k.a. a receiver)}
\nomenclature{${\bf s}$}{A source point in space (a.k.a. a source)}
\nomenclature{${\bf p}$}{A parameter of the space (e.g. density, conductivity, etc.)}
\nomenclature{${\rm d}p$}{A differential with respect to $p$}
\nomenclature{${\rm d}_p$}{Derivative operator with respect to $p$. It is a contraction of $\frac{{\rm d}}{{\rm d}p}$}
\nomenclature{$\partial_m$}{A partial derivative with respect to $m$}
\nomenclature{$\nabla_{\bf p} = (\partial_{p_{1}}, \dots, \partial_{p_{n}})$}{A gradient operator (row vector) with respect to ${\bf p}$}
\nomenclature{$\nabla^2$}{The Laplacian operator}
\nomenclature{$\mathcal{L}$}{Differential operator (continuous)}
\nomenclature{$\tau = \tau( {\bf x}; \, {\bf p} )$}{Travel time}

\nomenclature{$u^o = u( {\bf x}, \, t; \, \pvec )$}{The natural wavefield in the time domain}
\nomenclature{$u_r^o = u( \xr, \, t; \, {\bf p} )$}{Observed wave field at receiver due to source}
\nomenclature{$u = u( {\bf x}, \, t; \, {\bf p} )$}{The synthetic wavefield in the time domain}
\nomenclature{$u_r = u( {\bf x}_r, \, t; \, {\bf p} )$}{Synthetic wave field at receiver due to source}

% \nomenclature{$u^o= u^o( {\bf r},\, t; \, {\bf p})$}{}
% \nomenclature{$u_{N}$}{Observed wave field in actual experiment (Nature)}
\nomenclature{$\tilde{u} = \tilde{u}( {\bf x}, \, \omega; \, {\bf p} )$}{A wavefield in the frequency domain}

\nomenclature{$\bar{a}$}{Overbar on any variable indicates complex conjugation}
% \nomenclature{$M^t$}{Transpose}
% \nomenclature{$M^* = \bar{M}^t$}{Hermitian}
\nomenclature{$e=u^o - u_{N}^{o}$}{Error}
\nomenclature{${\sf E = E}({\bf p}, \, u^o; \, u_{N}^{o})$}{Objective function (aka {\sf E}nergy, see Demanet)}
